    <script>
        // Toggle Mobile Menu
        function toggleMobileMenu() {
            const menu = document.getElementById('mobile-menu');
            menu.classList.toggle('hidden');
        }

        // Modal Logic
        function openModal(packageName) {
            const modal = document.getElementById('quote-modal');
            const nameSpan = document.getElementById('modal-package-name');
            const hiddenInput = document.getElementById('modal-hidden-package');
            
            nameSpan.textContent = packageName;
            hiddenInput.value = packageName;
            
            modal.classList.remove('hidden');
        }

        function closeModal() {
            document.getElementById('quote-modal').classList.add('hidden');
        }

        // Close modal on outside click
        window.onclick = function(event) {
            const modal = document.getElementById('quote-modal');
            if (event.target == modal) {
                closeModal();
            }
        }

        // Form Submission Logic (Simulation)
        function submitModalForm(e) {
            e.preventDefault();
            closeModal();
            showToast();
        }

        function submitCustomForm(e) {
            e.preventDefault();
            showToast();
        }

        function showToast() {
            const toast = document.getElementById('success-toast');
            toast.classList.remove('translate-x-full');
            setTimeout(() => {
                toast.classList.add('translate-x-full');
            }, 4000);
        }
    </script>
